    **Gate 22 — gallery-order parity walk.** Catches Pitfall P-1. For each populated multi-piece category:
      1. Extract gallery tile order from `dist/<cat>/index.html` — every `<a href="/<cat>/<slug>"` occurrence in document order.
      2. Walk the next-chain starting from the first slug's detail page: read its `class="pager-link next" href="/<cat>/<nextslug>"`, then load that next detail page, repeat until next is absent.
      3. The slug sequence from the walk MUST equal the gallery tile order.

    ```bash
    # Gate 22: gallery-order parity — prev/next chain matches gallery tile order (Pitfall P-1).
    for cat in design finance personal marketing; do
      [[ -f "$DIST/$cat/index.html" ]] || continue

      # 1. Extract gallery tile slug order. The gallery uses <a href="/<cat>/<slug>"> for each piece.
      # Use sort -u-preserving extraction: walk document order, dedup contiguous repeats only.
      gallery_slugs=$(grep -oE 'href="/'"$cat"'/[a-z0-9-]+"' "$DIST/$cat/index.html" 2>/dev/null \
                      | sed -E 's|href="/'"$cat"'/||; s|"$||' \
                      | awk '!seen[$0]++' || true)
      if [[ -z "$gallery_slugs" ]]; then
        echo "  OK: $cat — empty gallery or no pieces extractable; skipping parity walk"
        continue
      fi

      gallery_count=$(echo "$gallery_slugs" | wc -l | tr -d ' ')
      if [[ "$gallery_count" -le 1 ]]; then
        echo "  OK: $cat — single piece, parity walk vacuous"
        continue
      fi

      # 2. Walk next-chain from first slug.
      first_slug=$(echo "$gallery_slugs" | head -1)
      walk=()
      current="$first_slug"
      max_iters=20  # safety cap (Pitfall: should never loop, but guard against future bugs)
      while [[ -n "$current" && ${#walk[@]} -lt $max_iters ]]; do
        walk+=("$current")
        detail="$DIST/$cat/$current/index.html"
        [[ -f "$detail" ]] || break
        # Extract next-pager href
        next_href=$(grep -oE '<a [^>]*class="pager-link next"[^>]*href="[^"]+"' "$detail" 2>/dev/null \
                    | grep -oE 'href="[^"]+"' | sed 's/href="//;s/"$//' | head -1 || true)
        if [[ -z "$next_href" ]]; then break; fi
        # Strip leading /<cat>/ and trailing slash to get next slug
        next_slug=$(echo "$next_href" | sed -E "s|^/$cat/||; s|/$||")
        if [[ "$next_slug" == "$current" ]]; then break; fi  # avoid self-loop
        current="$next_slug"
      done

      walk_sequence=$(printf '%s\n' "${walk[@]}")
      if [[ "$walk_sequence" != "$gallery_slugs" ]]; then
        echo "  FAIL: $cat — pager next-chain diverges from gallery order (Pitfall P-1)"
        echo "    gallery: $(echo "$gallery_slugs" | tr '\n' ' ')"
        echo "    pager:   $(echo "$walk_sequence" | tr '\n' ' ')"
        fail=1
      else
        echo "  OK: $cat — pager next-chain matches gallery tile order ($gallery_count pieces)"
      fi
    done
    ```
